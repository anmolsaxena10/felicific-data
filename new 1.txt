#include<stdio.h>
#include<string.h>

#define RETAIN 0
#define REPLACE 1
#define DEL 2
#define INS 3



int main()
{
	char s1[] = "hogwarts";
	char s2[] = "hawards";
	int i,j,tmp,l1=strlen(s1),l2=strlen(s2);
	int newDel,newIns,newRepl
	int add,del,repl;
	add = del = 1;
	repl = 2;
	
	int sol[l1+1][l2+1];
	for(i=0;i<=l1;i++)
	{
		sol[i][0] = i;
		edit[i][0] = DEL;
	}
	for(j=0;j<=l2;j++)
	{
		sol[0][j] = j;
		edit[0][j] = INS;
	}
	
	for(i=1;i<=11;i++)
	{
		for(j=1;j<=l2;j++)
		{
			if( s1[i-1] == s2[j-1] )
			{
				sol[i][j] = sol[i-1][j-1];
				edit[i][j] = RETAIN;
			}
			else
			{
				newDel = sol[i-1][j] + del;
				newIns = sol[i][j-1] + add;
				newRepl = sol[i-1][j-1] + repl;
			
				if(newDel < newIns && newDel < newRepl)
				{
					sol[i][j] = newDel;
					edit[i][j] = DEL;
				}
				else if(newIns < newDel && newIns < newRepl)
				{
					sol[i][j] = newIns;
					edit[i][j] = INS;
				}
				else
				{
					sol[i][j] = newRepl;
					edit[i][j] = REPLACE;
				}
			}
		}
	}
	
	printf("Minimum edit distance: %d",sol[l1][l2]);

}
